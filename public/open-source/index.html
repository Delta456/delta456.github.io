Limited Operator Overloading#

struct Vec {
    x int
    y int
}

fn (a Vec) str() string {
    return '{${a.x}, ${a.y}}'
}

fn (a Vec) + (b Vec) Vec {
    return Vec{a.x + b.x, a.y + b.y}
}

fn (a Vec) - (b Vec) Vec {
    return Vec{a.x - b.x, a.y - b.y}
}

fn main() {
    a := Vec{2, 3}
    b := Vec{4, 5}
    mut c := Vec{1, 2}

    println(a + b) // "{6, 8}"
    println(a - b) // "{-2, -2}"
    c += a
    //^^ autogenerated from + overload
    println(c) // "{3, 5}"
}

Implemented in PRs #7737, #774, 8030, 8067, #8086, #9529 and #9529

Match with Const Expr#

const start = 1

const end = 10

c := 2
num := match c {
    start...end {
        1000
    }
    else {
        0
    }
}
println(num)
// 1000

Implemented in PRs #16526 and #19572
